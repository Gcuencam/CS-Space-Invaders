#!groovy
pipeline {
node{
    checkout scm
    agent none
    stages {
        stage('Build') {
            agent any
            steps {
                checkout scm
                bat 'make'
                stash includes: '**/target/*.jar', name: 'Space Invaders' 
            }
        }
        stage('Test') {
            agent {
                label 'windows'
            }
            steps {
                unstash 'app'
                bat 'make check' 
            }
            post {
                always {
                    junit '**/target/*.xml'
                }
            }
        }
    }
}
}
